"scripts": {
  "test": "mocha \"src/tests/**/*.spec.js\" --timeout 30000",
  "smoke": "mocha \"src/tests/smoke/**/*.spec.js\" --timeout 30000",
  "contract": "mocha \"src/tests/contract/**/*.spec.js\" --timeout 30000",
  "integration": "mocha \"src/tests/integration/**/*.spec.js\" --timeout 60000",
  "regression": "mocha \"src/tests/regression/**/*.spec.js\" --timeout 60000"
}

issue: 


Error [ERR_REQUIRE_ESM]: require() of ES

fix add "type": "module" inside package.json

{
  "name": "api-automation",
  "version": "1.0.0",
  "type": "module",
  "scripts": {
    "smoke": "mocha \"src/tests/smoke/**/*.spec.js\" --timeout 30000"
  }
}

replace require : 

const { expect } = require("chai");
const { apiGet } = require("../../client/httpClient");

after fix:

const { expect } = require("chai");
const { apiGet } = require("../../client/httpClient");

require() = CommonJS

import = ES Modules

So once you switch project to ESM, you must change:

const x = require("x");
module.exports = ...

to 

import x from "x";
export ...
